#include "edx-io.hpp"
#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

long int n;
vector<int> mas;

void input()
{
    io >> n;
    mas.resize(n);
    for (long int i = 0; i < n; i++)
        mas[i] = i + 1;
}

void solve()
{
    for (long int i = 2; i < n; i++)
        swap(mas[i], mas[i / 2]);
}

void output()
{
    for (long int i = 0; i < n; i++)
        io << mas[i] << " ";
}

int main(int argc, char const *argv[])
{
    input();
    solve();
    output();
    return 0;
}

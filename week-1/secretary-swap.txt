#include "edx-io.hpp"
#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n;
    io >> n;
    int max = INT32_MIN;
    int max_pos = 0;
    int min = INT32_MAX;
    int min_pos = 0;

    int *x = new int[n];

    for (int i = 0; i < n; i++)
    {
        io >> x[i];

        if (x[i] < min)
        {
            min = x[i];
            min_pos = i;
        }
        if (x[i] > max)
        {
            max = x[i];
            max_pos = i;
        }
    }

    /*if (min_pos > 0)
    {
        swap(x[min_pos], x[0]);
        io << "Swap elements at indices " << 1 << " and " << min_pos + 1 << ".\n";
    }
    if (max_pos == 0)
        max_pos = min_pos;
    if (max_pos < n - 1)
    {
        swap(x[max_pos], x[n - 1]);
        io << "Swap elements at indices " << max_pos + 1 << " and " << n << ".\n";
    }*/

    for (int i = 1; i < n; i++)
    {
        int k = 0;
        for (int j = i; j > 0 && x[j - 1] > x[j]; j--)
        {
            swap(x[j - 1], x[j]);
            k++;
        }
        if (i < i + k - 1)
            io << "Swap elements at indices " << i << " and " << i + k - 1 << ".\n";
        k = 0;
    }

    io << "No more swaps needed.\n";
    for (int i = 0; i < n; ++i)
        io << x[i] << ' ';

    return 0;
}
